import { test, expect, type Page } from '@playwright/test'
import { createClient } from '@supabase/supabase-js'

// Cliente Supabase admin para setup de testes (usa service role key)
const supabaseAdmin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL ?? 'http://localhost:54321',
  process.env.SUPABASE_SERVICE_ROLE_KEY ?? '',
  { auth: { persistSession: false } }
)

const TEST_USER_EMAIL = 'test-user@open-ledger.local'
const TEST_USER_PASSWORD = 'test-password-123'

// Helper: autenticar usuário de teste via Supabase API e injetar cookies na página
async function loginTestUser(page: Page) {
  // Fazer login via Supabase API (não via UI — evita Google OAuth)
  const { data, error } = await supabaseAdmin.auth.signInWithPassword({
    email: TEST_USER_EMAIL,
    password: TEST_USER_PASSWORD,
  })

  if (error || !data.session) {
    throw new Error(`Falha ao autenticar usuário de teste: ${error?.message ?? 'sem session'}`)
  }

  // Injetar session como cookies na página (simula login completo)
  await page.goto('/login')
  await page.evaluate(
    async ({ accessToken, refreshToken }) => {
      // Criar cliente browser e setar session diretamente
      const { createBrowserClient } = await import('@supabase/ssr')
      const supabase = createBrowserClient(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
      )
      await supabase.auth.setSession({ access_token: accessToken, refresh_token: refreshToken })
    },
    {
      accessToken: data.session.access_token,
      refreshToken: data.session.refresh_token,
    }
  )

  // Navegar para o app após injetar session
  await page.goto('/profile')
  await page.waitForURL('**/profile')
}

**2. Criar `tests/e2e/rls.spec.ts`** — testa AUTH-03 (isolamento de dados):

```typescript
import { test, expect } from '@playwright/test'
import { createClient } from '@supabase/supabase-js'

// Cliente Supabase admin para operações privilegiadas nos testes
const supabaseAdmin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL ?? 'http://localhost:54321',
  process.env.SUPABASE_SERVICE_ROLE_KEY ?? '',
  { auth: { persistSession: false } }
)

const TEST_USER_1_EMAIL = 'test-user@open-ledger.local'
const TEST_USER_1_PASSWORD = 'test-password-123'
const TEST_USER_2_EMAIL = 'test-user-2@open-ledger.local'
const TEST_USER_2_PASSWORD = 'test-password-123'

// AUTH-03: RLS impede usuário 1 de acessar dados do usuário 2
test('@smoke @auth-03 - RLS: usuário 1 não consegue ler dados do usuário 2', async () => {
  // Obter IDs dos usuários via admin
  const { data: users } = await supabaseAdmin.auth.admin.listUsers()
  const user1 = users.users.find((u) => u.email === TEST_USER_1_EMAIL)
  const user2 = users.users.find((u) => u.email === TEST_USER_2_EMAIL)

  if (!user1 || !user2) {
    throw new Error('Usuários de teste não encontrados — rodar supabase db reset primeiro')
  }

  // Inserir uma account para o usuário 2 via admin (bypass RLS para setup)
  const { data: account, error: insertError } = await supabaseAdmin
    .from('accounts')
    .insert({
      user_id: user2.id,
      name: 'Conta Secreta User 2',
      type: 'checking',
      color: '#ff0000',
    })
    .select()
    .single()

  if (insertError) {
    throw new Error(`Falha ao criar account de teste: ${insertError.message}`)
  }

  const accountId = account.id

  try {
    // Autenticar como usuário 1
    const { data: session1 } = await supabaseAdmin.auth.signInWithPassword({
      email: TEST_USER_1_EMAIL,
      password: TEST_USER_1_PASSWORD,
    })

    if (!session1.session) {
      throw new Error('Falha ao obter session do usuário 1')
    }

    // Criar cliente com token do usuário 1 (sujeito a RLS)
    const supabaseUser1 = createClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL ?? 'http://localhost:54321',
      process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY ?? '',
      {
        global: {
          headers: { Authorization: `Bearer ${session1.session.access_token}` },
        },
      }
    )

    // Usuário 1 NÃO deve conseguir ver a account do usuário 2
    const { data: result } = await supabaseUser1
      .from('accounts')
      .select('*')
      .eq('id', accountId)

    // RLS deve retornar array vazio — não a account do user2
    expect(result).toEqual([])
  } finally {
    // Cleanup: remover account de teste do user2
    await supabaseAdmin.from('accounts').delete().eq('id', accountId)
  }
})

// AUTH-03: Usuário consegue ler seus próprios dados
test('@smoke @auth-03 - RLS: usuário consegue ler seus próprios dados', async () => {
  const { data: session1 } = await supabaseAdmin.auth.signInWithPassword({
    email: TEST_USER_1_EMAIL,
    password: TEST_USER_1_PASSWORD,
  })

  if (!session1.session) {
    throw new Error('Falha ao obter session do usuário 1')
  }

  const supabaseUser1 = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL ?? 'http://localhost:54321',
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY ?? '',
    {
      global: {
        headers: { Authorization: `Bearer ${session1.session.access_token}` },
      },
    }
  )

  // Usuário 1 deve conseguir ver seu próprio profile
  const { data: profiles, error } = await supabaseUser1
    .from('profiles')
    .select('id')

  expect(error).toBeNull()
  expect(profiles).toHaveLength(1)
  expect(profiles![0].id).toBe(session1.session.user.id)
})
```
