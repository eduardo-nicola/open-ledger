```typescript
import { defineConfig, devices } from '@playwright/test'
import path from 'path'

const authFile = path.join(__dirname, 'tests/.auth/user.json')

/**
 * Configuração Playwright para Open Ledger E2E tests.
 * Testes rodam contra o app Next.js local (http://localhost:3000)
 * com Supabase local (supabase start deve estar rodando).
 *
 * Projetos:
 * - setup: cria o estado de autenticação (tests/.auth/user.json)
 * - chromium: testes que requerem autenticação (usa storageState do setup)
 * - chromium-unauth: testes que requerem contexto não-autenticado (sem storageState)
 */
export default defineConfig({
  testDir: './tests/e2e',
  timeout: 30 * 1000,          // 30s por teste
  expect: { timeout: 5000 },
  fullyParallel: false,        // serial para evitar conflitos de sessão
  forbidOnly: !!process.env.CI,
  retries: process.env.CI ? 2 : 0,
  reporter: [['html', { open: 'never' }], ['list']],

  use: {
    baseURL: 'http://localhost:3000',
    trace: 'on-first-retry',
    screenshot: 'only-on-failure',
  },

  projects: [
    {
      name: 'setup',
      testMatch: '**/auth.setup.ts',
    },
    {
      name: 'chromium',
      use: {
        ...devices['Desktop Chrome'],
        storageState: authFile,
      },
      dependencies: ['setup'],
    },
    {
      name: 'chromium-unauth',
      use: { ...devices['Desktop Chrome'] },
      // Sem storageState — contexto não-autenticado para testar redirects
    },
  ],

  // Iniciar servidor Next.js antes dos testes
  webServer: {
    command: 'npm run dev',
    url: 'http://localhost:3000',
    reuseExistingServer: !process.env.CI,
    timeout: 60 * 1000,
  },
})
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Test code → Supabase Admin API | Service role key usada nos testes — bypass de RLS para setup; nunca em código de produção |
| seed.sql → auth.users | Inserção direta em tabela interna do Supabase — apenas em dev local |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Information Disclosure | Service role key em test files | mitigate | Service role key lida de env var (`process.env.SUPABASE_SERVICE_ROLE_KEY`), nunca hardcoded; .env.local no .gitignore; testes só rodam localmente com dev Supabase |
| T-05-02 | Information Disclosure | Credenciais reais em seed.sql | mitigate | seed.sql usa apenas usuários fictícios locais (`test-user@open-ledger.local`) — senha bcrypt de test-password-123; nenhum dado real |
| T-05-03 | Elevation of Privilege | Testes com service role bypassando RLS | accept | Uso intencional para setup de fixtures de teste; service role NÃO está no código de produção; rls.spec.ts verifica o isolamento com anon key (sujeito a RLS) |
</threat_model>
