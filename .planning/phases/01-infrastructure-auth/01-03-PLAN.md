```typescript
import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'
import type { Database } from '@/types/database'

export async function createClient() {
  const cookieStore = await cookies()

  return createServerClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return cookieStore.getAll()
        },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            )
          } catch {
            // Ignorado em Server Components — o middleware faz o refresh dos cookies
          }
        },
      },
    }
  )
}
```

**1. Criar `middleware.ts`** na raiz do projeto (mesmo nível de `package.json`):

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'

export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({
    request,
  })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll()
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) =>
            request.cookies.set(name, value)
          )
          supabaseResponse = NextResponse.next({
            request,
          })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          )
        },
      },
    }
  )

  // CRÍTICO: usar getUser() — não getSession() — para validar JWT no servidor (Pitfall 1)
  // getUser() faz round-trip à API Supabase para validação criptográfica do token
  const {
    data: { user },
  } = await supabase.auth.getUser()

  const isAuthRoute = request.nextUrl.pathname.startsWith('/login')
  const isCallbackRoute = request.nextUrl.pathname.startsWith('/auth')

  // Redirecionar usuário não autenticado para /login
  if (!user && !isAuthRoute && !isCallbackRoute) {
    const url = request.nextUrl.clone()
    url.pathname = '/login'
    return NextResponse.redirect(url)
  }

  // Redirecionar usuário autenticado que acessa /login para /profile
  if (user && isAuthRoute) {
    const url = request.nextUrl.clone()
    url.pathname = '/profile'
    return NextResponse.redirect(url)
  }

  // CRÍTICO: retornar supabaseResponse (não NextResponse.next()) para preservar cookies renovados
  return supabaseResponse
}

export const config = {
  matcher: [
    /*
     * Aplica middleware em todos os paths exceto:
     * - _next/static (arquivos estáticos)
     * - _next/image (otimização de imagens)
     * - favicon.ico
     * - arquivos com extensão (ex: .svg, .png, .jpg)
     */
    '/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
}
```

**2. Criar `app/auth/callback/route.ts`** — Route Handler PKCE para Google OAuth:

```typescript
import { NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'

export async function GET(request: Request) {
  const { searchParams, origin } = new URL(request.url)
  const code = searchParams.get('code')
  const next = searchParams.get('next') ?? '/profile'

  if (code) {
    const supabase = await createClient()
    const { error } = await supabase.auth.exchangeCodeForSession(code)

    if (!error) {
      // Redirecionar para o destino após autenticação bem-sucedida
      return NextResponse.redirect(`${origin}${next}`)
    }
  }

  // Redirecionar para /login com indicador de erro se algo falhar
  return NextResponse.redirect(`${origin}/login?error=auth_callback_failed`)
}
```

**NOTAS IMPORTANTES:**
- O middleware usa `createServerClient` diretamente (não `createClient()` de lib/supabase/server.ts) porque precisa acesso ao `request` e `supabaseResponse` para o padrão de cookies correto
- O callback route usa `createClient()` de `@/lib/supabase/server` normalmente
- O parâmetro `next` permite deep-links futuros (ex: redirecionar para página que o usuário queria acessar)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Next.js middleware | Todo request externo cruza aqui — ponto de enforcement de auth |
| Google OAuth → /auth/callback | Code PKCE chega via redirect — deve ser trocado atomicamente |
| middleware → Supabase Auth API | getUser() valida JWT criptograficamente (round-trip) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | `getSession()` no servidor | mitigate | Banido: sempre `getUser()` no middleware e server components; `getUser()` faz round-trip à API Supabase para validação criptográfica do JWT — `getSession()` lê cookies sem verificar assinatura |
| T-03-02 | Spoofing | JWT token forjado | mitigate | `getUser()` valida token contra a API Supabase — token forjado retorna null, triggering redirect para /login |
| T-03-03 | Tampering | Session expira sem renovação | mitigate | Middleware OBRIGATÓRIO com padrão exato `getAll/setAll` — sem retornar `supabaseResponse`, cookies renovados são perdidos (Pitfall 2) |
| T-03-04 | Tampering | OAuth state parameter manipulation (CSRF) | accept | PKCE flow (code_verifier/challenge SHA-256) implementado pelo Supabase — mitiga CSRF em OAuth; Supabase gerencia code_verifier automaticamente |
| T-03-05 | Tampering | Open redirect em /auth/callback | mitigate | Parâmetro `next` é limitado a paths relativos ao `origin` — `${origin}${next}` evita redirect para domínio externo |
</threat_model>
