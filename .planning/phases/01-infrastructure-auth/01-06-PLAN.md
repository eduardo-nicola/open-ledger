```typescript
// tests/e2e/auth.setup.ts
// Autenticação via storageState para testes Playwright.
// Credencial: test@open-ledger.local (criada no supabase/seed.sql)
import { createClient } from '@supabase/supabase-js'
import { test as setup } from '@playwright/test'
import path from 'path'

const authFile = path.join(__dirname, '../.auth/user.json')

const TEST_EMAIL = 'test@open-ledger.local'
const TEST_PASSWORD = 'test-password-open-ledger-123'

setup('autenticar usuário de teste e salvar storageState', async ({ page }) => {
  const adminClient = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    { auth: { autoRefreshToken: false, persistSession: false } }
  )

  // Garantir que o usuário de teste existe (seed.sql deve ter criado, mas por segurança)
  try {
    await adminClient.auth.admin.createUser({
      email: TEST_EMAIL,
      password: TEST_PASSWORD,
      email_confirm: true,
      user_metadata: {
        full_name: 'Test User',
        avatar_url: null,
      },
    })
  } catch {
    // Usuário já existe — continuar normalmente
  }

  // Autenticar via API (não via UI — evita Google OAuth)
  const anonClient = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  )

  const { data, error } = await anonClient.auth.signInWithPassword({
    email: TEST_EMAIL,
    password: TEST_PASSWORD,
  })

  if (error || !data.session) {
    throw new Error(`auth.setup.ts falhou: ${error?.message ?? 'sem sessão'}`)
  }

  // Navegar para o app e injetar session via localStorage
  await page.goto('/login')
  await page.evaluate(
    ({ url, accessToken, refreshToken }) => {
      const key = `sb-${new URL(url).hostname}-auth-token`
      localStorage.setItem(
        key,
        JSON.stringify({
          access_token: accessToken,
          refresh_token: refreshToken,
          token_type: 'bearer',
          expires_in: 3600,
          expires_at: Math.floor(Date.now() / 1000) + 3600,
        })
      )
    },
    {
      url: process.env.NEXT_PUBLIC_SUPABASE_URL!,
      accessToken: data.session.access_token,
      refreshToken: data.session.refresh_token,
    }
  )

  // Aguardar que o middleware reconheça a sessão e redirecione para /profile
  await page.reload()
  await page.waitForURL('**/profile', { timeout: 10000 })

  // Persistir estado de autenticação para os testes subsequentes
  await page.context().storageState({ path: authFile })
})
```

```typescript
import { test, expect } from '@playwright/test'
import { createClient } from '@supabase/supabase-js'

// =============================================================================
// rls.spec.ts — Testes de isolamento de dados RLS (AUTH-03)
// Sem storageState — usa Supabase client diretamente com JWT
// Credencial 1: test@open-ledger.local
// Credencial 2: test-2@open-ledger.local
// =============================================================================

const supabaseAdmin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL ?? 'http://localhost:54321',
  process.env.SUPABASE_SERVICE_ROLE_KEY ?? '',
  { auth: { persistSession: false } }
)

const TEST_USER_1_EMAIL = 'test@open-ledger.local'
const TEST_USER_1_PASSWORD = 'test-password-open-ledger-123'
const TEST_USER_2_EMAIL = 'test-2@open-ledger.local'

// AUTH-03: Usuário 1 não consegue ler dados do usuário 2
test('@smoke @auth-03 - RLS: usuário 1 não consegue ler dados do usuário 2', async () => {
  const { data: users } = await supabaseAdmin.auth.admin.listUsers()
  const user2 = users.users.find((u) => u.email === TEST_USER_2_EMAIL)

  if (!user2) {
    throw new Error('Usuário 2 de teste não encontrado — rodar supabase db reset primeiro')
  }

  // Inserir account para o usuário 2 via admin (bypass RLS para setup)
  const { data: account, error: insertError } = await supabaseAdmin
    .from('accounts')
    .insert({
      user_id: user2.id,
      name: 'Conta Secreta User 2',
      type: 'checking',
      color: '#ff0000',
    })
    .select()
    .single()

  if (insertError) {
    throw new Error(`Falha ao criar account de teste: ${insertError.message}`)
  }

  const accountId = account.id

  try {
    const { data: session1 } = await supabaseAdmin.auth.signInWithPassword({
      email: TEST_USER_1_EMAIL,
      password: TEST_USER_1_PASSWORD,
    })

    if (!session1.session) throw new Error('Falha ao obter session do usuário 1')

    const supabaseUser1 = createClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL ?? 'http://localhost:54321',
      process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY ?? '',
      { global: { headers: { Authorization: `Bearer ${session1.session.access_token}` } } }
    )

    const { data: result } = await supabaseUser1.from('accounts').select('*').eq('id', accountId)
    expect(result).toEqual([]) // RLS deve retornar array vazio
  } finally {
    await supabaseAdmin.from('accounts').delete().eq('id', accountId)
  }
})

// AUTH-03: Usuário consegue ler seus próprios dados
test('@smoke @auth-03 - RLS: usuário consegue ler seus próprios dados', async () => {
  const { data: session1 } = await supabaseAdmin.auth.signInWithPassword({
    email: TEST_USER_1_EMAIL,
    password: TEST_USER_1_PASSWORD,
  })

  if (!session1.session) throw new Error('Falha ao obter session do usuário 1')

  const supabaseUser1 = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL ?? 'http://localhost:54321',
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY ?? '',
    { global: { headers: { Authorization: `Bearer ${session1.session.access_token}` } } }
  )

  const { data: profiles, error } = await supabaseUser1.from('profiles').select('id')
  expect(error).toBeNull()
  expect(profiles).toHaveLength(1)
  expect(profiles![0].id).toBe(session1.session.user.id)
})
```
